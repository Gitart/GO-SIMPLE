## Float64 -> Str

```go
/******************************************************************************************************************************************************
 * Конвертация Float64 to Str (с 6 десятичными )
 ******************************************************************************************************************************************************/
func FloatToString(input_num float64) string {
	 return strconv.FormatFloat(input_num, 'f', 6, 64)
}
```

## Float64 -> Str
```go

/******************************************************************************************************************************************************
 * Конвертация Float64 to Str без десятичных
 ******************************************************************************************************************************************************/
func FloatToStr(input_num float64) string {
	 return strconv.FormatFloat(input_num, 'f', 0, 64)
}
```

```go
func isNil(a interface{}) bool {
     defer func() { recover() }()
     return a == nil || reflect.ValueOf(a).IsNil()
}
```

```go
/****************************************************************************************
 * Check null is interface

func ISNIL(Vars string) string{
     defer func() { recover() }()
     
     if len(Vars) == 0 || reflect.ValueOf(Vars).IsNil()  {
        return "33"
     }
     return Vars
}
```
