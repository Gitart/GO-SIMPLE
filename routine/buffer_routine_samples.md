    "fmt"
    "math/rand"
    "sync"
    "time"
)

    randomno int

    sumofdigits int

func digits(number int) int {
    sum := 0
    no := number
    for no != 0 {
        digit := no % 10
        sum += digit
        no /= 10
    }

    return sum

    for job := range jobs {
        output := Result{job, digits(job.randomno)}

    }
    wg.Done()
}

        fmt.Printf("Job id %d, input random no %d , sum of digits %d\n", result.job.id, result.job.randomno, result.sumofdigits)
